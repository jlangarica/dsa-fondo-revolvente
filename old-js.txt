    /* --- ENLACE CON EL ORQUESTADOR AUTOMATIZADO (DRIVE + GEMINI API) --- */
    function procesarOficioConIA() {
      var fileInput = document.getElementById('pdfFile');
      if (!fileInput.files || fileInput.files.length === 0) {
        showToast('Seleccione un archivo PDF primero.', 'warning');
        return;
      }
      var file = fileInput.files[0];
      var reader = new FileReader();
      var btn = document.getElementById('btnGemini');
      btn.disabled = true;
      btn.innerHTML = '<div class="loading-spinner" style="width:16px;height:16px;border-width:2px;margin:0;border-top-color:white;"></div> Extrayendo con IA…';
      
      showToast('Sincronizando con Google Drive y procesando con Gemini...', 'neutral');

      reader.onload = function(e) {
        var base64Data = e.target.result.split(',')[1];
        
        // Obtenemos el nombre textual de la UC seleccionada actualmente como fallback
        var selectUcEl = document.getElementById('id_uc');
        var ucNombreFallback = selectUcEl.selectedIndex !== -1 ? selectUcEl.options[selectUcEl.selectedIndex].text : "Unidad_DSA";

        // Ejecutamos el orquestador nativo en el servidor (Crea carpeta Drive y corre Gemini)
        google.script.run
          .withSuccessHandler(function(response) {
            if (response.status === "ERROR") {
              showToast(response.error, 'error');
              btn.disabled = false;
              btn.innerHTML = 'Analizar con IA';
              return;
            }

            var payload = response.data;
            showToast('Expediente creado en Google Drive: ' + payload.folioGenerado, 'success');
            
            // Seteamos las variables globales de las referencias cruzadas de Drive
            currentDriveFolderId = payload.driveFolderId;
            currentDriveFileId = payload.driveFileId;

            // Rellenamos de inmediato el formulario
            if (payload.formularioMapeado.folio_dsa) document.getElementById('folio_dsa').value = payload.formularioMapeado.folio_dsa;
            if (payload.formularioMapeado.no_oficio) document.getElementById('no_oficio').value = payload.formularioMapeado.no_oficio;
            if (payload.formularioMapeado.fecha_recepcion) document.getElementById('fecha_recepcion').value = payload.formularioMapeado.fecha_recepcion;
            if (payload.formularioMapeado.observaciones) document.getElementById('observaciones').value = payload.formularioMapeado.observaciones;

            // Autocompletado reactivo de Unidad Solicitante (id_uc)
            if (payload.formularioMapeado.uc_cod_sugerido) {
              var selectUc = document.getElementById('id_uc');
              for (var i = 0; i < selectUc.options.length; i++) {
                if (selectUc.options[i].text.indexOf(payload.formularioMapeado.uc_cod_sugerido) !== -1) {
                  selectUc.selectedIndex = i;
                  break;
                }
              }
            }

            // RENDERIZADO DEL BADGE REACTIVO PARA LA NEGATIVA DE EXISTENCIA (DOC-02)
            var badgeContainer = document.getElementById('negativaBadgeContainer');
            if (payload.formularioMapeado.negativa_validada_previas) {
              badgeContainer.innerHTML = '<span class="badge badge-success" style="padding: 8px 12px;">Validada (' + payload.formularioMapeado.fecha_negativa + ')</span>';
              showToast('DOC-02: Negativa de Existencia validada cronológicamente.', 'success');
            } else {
              badgeContainer.innerHTML = '<span class="badge badge-danger" style="padding: 8px 12px;">No detectada en PDF</span>';
              showToast('Alerta: No se localizó la hoja de Negativa en el escaneo.', 'warning');
            }

            // Rellenamos la tabla dinámica de artículos
            if (payload.formularioMapeado.articulos && payload.formularioMapeado.articulos.length > 0) {
              document.getElementById('tablaDetalles').innerHTML = ''; // Limpiar mocks
              payload.formularioMapeado.articulos.forEach(function(art) {
                agregarFilaArticuloConValores(art.codigo_art || '', art.cantidad || 1, art.precio_unitario || '');
              });
            }

            btn.disabled = false;
            btn.innerHTML = '<svg class="icon-sm" viewBox="0 0 24 24"><path d="m12 3-1.912 5.813a2 2 0 0 1-1.275 1.275L3 12l5.813 1.912a2 2 0 0 1 1.275 1.275L12 21l1.912-5.813a2 2 0 0 1 1.275-1.275L21 12l-5.813-1.912a2 2 0 0 1-1.275-1.275L12 3Z"></path></svg> Analizar con IA y Crear Drive Folder';
          })
          .withFailureHandler(function(error) {
            showToast('Fallo en la comunicación con el servidor: ' + error.message, 'error');
            btn.disabled = false;
            btn.innerHTML = '<svg class="icon-sm" viewBox="0 0 24 24"><path d="m12 3-1.912 5.813a2 2 0 0 1-1.275 1.275L3 12l5.813 1.912a2 2 0 0 1 1.275 1.275L12 21l1.912-5.813a2 2 0 0 1 1.275-1.275L21 12l-5.813-1.912a2 2 0 0 1-1.275-1.275L12 3Z"></path></svg> Analizar con IA y Crear Drive Folder';
          })
          .procesarCargaOficioInicial(base64Data, ucNombreFallback);
      };
      reader.readAsDataURL(file);
    }

    /* --- LÓGICA DE TABLA DETALLES --- */
    function agregarFilaArticulo() { agregarFilaArticuloConValores('', 1, ''); }

    function agregarFilaArticuloConValores(codigo, cantidad, precio) {
      var tbody = document.getElementById('tablaDetalles');
      var row = document.createElement('tr');
      row.className = 'fila-articulo';
      var optCatalogo = ['<option value="">Seleccione…</option>'];
      cacheCatalogo.forEach(function(art) {
        var selected = art.codigo_art === codigo ? 'selected' : '';
        optCatalogo.push('<option value="' + art.codigo_art + '" ' + selected + '>' + art.codigo_art + ' - ' + art.descripcion + '</option>');
      });
      var optPacientes = ['<option value="">Ninguno / General</option>'];
      cachePacientes.forEach(function(pac) {
        optPacientes.push('<option value="' + pac.id_paciente + '">' + pac.nombre_completo + '</option>');
      });
      row.innerHTML = [
        '<td><select name="codigo_art" required onchange="detectarCambioArticulo(this)" class="table-select">' + optCatalogo.join('') + '</select></td>',
        '<td><input type="number" name="cantidad" min="1" value="' + (cantidad || 1) + '" required class="table-input"></td>',
        '<td><select name="id_paciente" class="table-select">' + optPacientes.join('') + '</select></td>',
        '<td><input type="number" step="0.01" name="precio_unitario" value="' + (precio || '') + '" required class="table-input"></td>',
        '<td><button type="button" class="btn btn-ghost" onclick="eliminarFila(this)"><svg class="icon-sm" viewBox="0 0 24 24"><polyline points="3 6 5 6 21 6"></polyline><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path></svg></button></td>'
      ].join('');
      tbody.appendChild(row);
    }

    function detectarCambioArticulo(selectElement) {
      var codigoSeleccionado = selectElement.value;
      var fila = selectElement.closest('tr');
      var inputPrecio = fila.querySelector('[name="precio_unitario"]');
      var articuloEncontrado = cacheCatalogo.find(function(art) { return art.codigo_art === codigoSeleccionado; });
      if (articuloEncontrado) { inputPrecio.value = articuloEncontrado.precio_sin_iva; } 
      else { inputPrecio.value = ''; }
    }

    function eliminarFila(btn) {
      var filas = document.getElementsByClassName('fila-articulo');
      if (filas.length > 1) { btn.closest('tr').remove(); } 
      else { showToast('El trámite debe contener al menos un artículo.', 'warning'); }
    }

    function limpiarFormulario() {
      document.getElementById('dsaForm').reset();
      document.getElementById('tablaDetalles').innerHTML = '';
      document.getElementById('negativaBadgeContainer').innerHTML = '<span class="badge badge-neutral" style="padding: 8px 12px;">Sin procesar archivo</span>';
      currentDriveFolderId = null;
      currentDriveFileId = null;
      agregarFilaArticulo();
      cerrarVistaPrevia();
    }

    /* --- ENVÍO DE FORMULARIO A BASE DE DATOS (SUPABASE) --- */
    function enviarFormulario(event) {
      event.preventDefault();
      var btn = document.getElementById('btnGuardar');
      btn.disabled = true;
      btn.innerHTML = '<div class="loading-spinner" style="width:16px;height:16px;border-width:2px;margin:0;border-top-color:white;"></div> Guardando…';

      var cabecera = {
        folio_dsa: document.getElementById('folio_dsa').value,
        id_tramite: document.getElementById('id_tramite').value ? Number(document.getElementById('id_tramite').value) : null,
        id_uc: document.getElementById('id_uc').value ? Number(document.getElementById('id_uc').value) : null,
        fecha_recepcion: document.getElementById('fecha_recepcion').value || null,
        id_servicio: document.getElementById('id_servicio').value ? Number(document.getElementById('id_servicio').value) : null,
        no_oficio: document.getElementById('no_oficio').value || null,
        id_usuario: document.getElementById('id_usuario').value ? Number(document.getElementById('id_usuario').value) : null,
        id_estatus_tramite: document.getElementById('id_estatus_tramite').value ? Number(document.getElementById('id_estatus_tramite').value) : null,
        observaciones: document.getElementById('observaciones').value || null,
        
        // INTEGRACIÓN SOBERANA DE DRIVE: Inyectamos los IDs de Google Drive en el payload relacional de Supabase
        drive_folder_id: currentDriveFolderId,
        drive_file_id: currentDriveFileId
      };

      var detalles = [];
      var filas = document.getElementsByClassName('fila-articulo');
      for (var i = 0; i < filas.length; i++) {
        var fila = filas[i];
        detalles.push({
          codigo_art: fila.querySelector('[name="codigo_art"]').value,
          cantidad: Number(fila.querySelector('[name="cantidad"]').value),
          id_paciente: fila.querySelector('[name="id_paciente"]').value ? Number(fila.querySelector('[name="id_paciente"]').value) : null,
          precio_unitario: Number(fila.querySelector('[name="precio_unitario"]').value)
        });
      }

      google.script.run
        .withSuccessHandler(function(folioConfirmado) {
          showToast('Expediente y Sincronización guardados en Supabase. Folio: ' + folioConfirmado, 'success');
          limpiarFormulario();
          btn.disabled = false;
          btn.innerHTML = '<svg class="icon-sm" viewBox="0 0 24 24"><path d="M19 21H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h11l5 5v11a2 2 0 0 1-2 2z"></path><polyline points="17 21 17 13 7 13 7 21"></polyline><polyline points="7 3 7 8 15 8"></polyline></svg> Guardar en Supabase';
          cargarDashboardView();
        })
        .withFailureHandler(function(error) {
          showToast('Error de inserción en Supabase: ' + error.message, 'error');
          btn.disabled = false;
          btn.innerHTML = '<svg class="icon-sm" viewBox="0 0 24 24"><path d="M19 21H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h11l5 5v11a2 2 0 0 1-2 2z"></path><polyline points="17 21 17 13 7 13 7 21"></polyline><polyline points="7 3 7 8 15 8"></polyline></svg> Guardar en Supabase';
        })
        .guardarCompraDesdeForm({ cabecera: cabecera, detalles: detalles });
    }
